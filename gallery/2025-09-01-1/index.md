<script>
// Ïù¥ÎØ∏ÏßÄ Î™©Î°ù Ï†ÄÏû•
let galleryImages = [];
let currentImageIndex = 0;

// ÌéòÏù¥ÏßÄ Î°úÎìú Ïãú Ïù¥ÎØ∏ÏßÄ Î™©Î°ù ÏàòÏßë
document.addEventListener('DOMContentLoaded', function() {
  const imageItems = document.querySelectorAll('.image-item');
  imageItems.forEach((item, index) => {
    const img = item.querySelector('img');
    if (img) {
      galleryImages.push(img.src);
    }
  });
});

// ÎùºÏù¥Ìä∏Î∞ïÏä§ Ïó¥Í∏∞
function openLightbox(imageSrc, index) {
  currentImageIndex = index;
  const lightbox = document.getElementById('lightbox');
  const lightboxImage = document.getElementById('lightboxImage');
  
  lightboxImage.src = imageSrc;
  lightbox.classList.add('show');
  document.body.style.overflow = 'hidden';
  
  // ÎÑ§ÎπÑÍ≤åÏù¥ÏÖò Î≤ÑÌäº ÌëúÏãú/Ïà®ÍπÄ
  updateNavButtons();
}

// ÎùºÏù¥Ìä∏Î∞ïÏä§ Îã´Í∏∞
function closeLightbox(event) {
  // Ïù¥Î≤§Ìä∏Í∞Ä Î≤ÑÌäºÏù¥ÎÇò Ïù¥ÎØ∏ÏßÄÏóêÏÑú Î∞úÏÉùÌïú Í≤ΩÏö∞ Î¨¥Ïãú
  if (event && event.target && (
    event.target.classList.contains('lightbox-image') ||
    event.target.classList.contains('lightbox-nav')
  )) {
    return;
  }
  
  const lightbox = document.getElementById('lightbox');
  lightbox.classList.remove('show');
  document.body.style.overflow = '';
}

// Ïù¥ÎØ∏ÏßÄ ÎÑ§ÎπÑÍ≤åÏù¥ÏÖò
function navigateLightbox(event, direction) {
  event.stopPropagation();
  
  currentImageIndex += direction;
  
  // Ïù∏Îç±Ïä§ Î≤îÏúÑ Ï≤¥ÌÅ¨
  if (currentImageIndex < 0) {
    currentImageIndex = galleryImages.length - 1;
  } else if (currentImageIndex >= galleryImages.length) {
    currentImageIndex = 0;
  }
  
  const lightboxImage = document.getElementById('lightboxImage');
  lightboxImage.src = galleryImages[currentImageIndex];
  
  updateNavButtons();
}

// ÎÑ§ÎπÑÍ≤åÏù¥ÏÖò Î≤ÑÌäº ÏóÖÎç∞Ïù¥Ìä∏
function updateNavButtons() {
  const prevBtn = document.querySelector('.lightbox-prev');
  const nextBtn = document.querySelector('.lightbox-next');
  
  // Ïù¥ÎØ∏ÏßÄÍ∞Ä 1Í∞úÎøêÏù¥Î©¥ ÎÑ§ÎπÑÍ≤åÏù¥ÏÖò Î≤ÑÌäº Ïà®ÍπÄ
  if (galleryImages.length <= 1) {
    prevBtn.style.display = 'none';
    nextBtn.style.display = 'none';
  } else {
    prevBtn.style.display = 'flex';
    nextBtn.style.display = 'flex';
  }
}

// ESC ÌÇ§Î°ú Îã´Í∏∞
document.addEventListener('keydown', function(e) {
  if (e.key === 'Escape') {
    const lightbox = document.getElementById('lightbox');
    if (lightbox.classList.contains('show')) {
      closeLightbox(e);
    }
  }
  
  // ÌôîÏÇ¥Ìëú ÌÇ§Î°ú ÎÑ§ÎπÑÍ≤åÏù¥ÏÖò
  if (e.key === 'ArrowLeft') {
    const lightbox = document.getElementById('lightbox');
    if (lightbox.classList.contains('show')) {
      navigateLightbox(e, -1);
    }
  }
  
  if (e.key === 'ArrowRight') {
    const lightbox = document.getElementById('lightbox');
    if (lightbox.classList.contains('show')) {
      navigateLightbox(e, 1);
    }
  }
});
</script>
